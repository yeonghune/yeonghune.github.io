## 버블, 선택, 삽입 정렬 코드 작성

### 버블

```c++
#include<iostream>
#define SWAP(x,y,t) ((t)=(x),(x)=(y),(y)=(t))
using namespace std;

int main(){
	int N;
	cin >> N;
	const int n = N;
	int arr[n] = {};
	
	for(int i = 0 ; i < N ; i++){
		int j;
		cin >> j;
		arr[i] = j;
	}
	
	for(int i =N-1; i >=0; i--){
		for( int j = 0; j < i; j++){
			int temp;
			if(arr[j] > arr[j+1]){
				SWAP(arr[j],arr[j+1],temp);
			}
		}
	}
	
	for(int i : arr) cout << i<<endl;
}
```

### 선택

```c++
#include<iostream>
#define SWAP(x,y,t) ((t)=(x),(x)=(y),(y)=(t))
using namespace std;

int main() {
	int N;
	cin >> N;
	int arr[1000] = {};

	for (int i = 0; i < N; i++) {
		int j;
		cin >> j;
		arr[i] = j;
	}

	for (int i = 0; i < N - 1; i++) {
		int temp, selection;
		int Min = 1001;
		for (int j = i + 1; j < N;j++) {
			if (arr[j] < Min) {
				selection = j;
				Min = arr[j];
			}
		}
		if ( arr[i] > arr[selection])SWAP(arr[i], arr[selection], temp);
}

	for (int i = 0; i < N; i++) cout << arr[i] << endl;
}
```

### 삽입

```c++
#include<iostream>
#define SWAP(x,y,t) ((t)=(x),(x)=(y),(y)=(t))
using namespace std;

int main() {
	int N;
	cin >> N;
	int arr[1000] = {};

	for (int i = 0; i < N; i++) {
		int j;
		cin >> j;
		arr[i] = j;
	}

	for (int i = 1; i < N; i++) {
		int j;
		int key = arr[i];
		for (j = i - 1; j >= 0 && arr[j] > key; j--) {
			arr[j + 1] = arr[j]; 
		}

		arr[j + 1] = key;
	}

	for (int i : arr) 
		if(i != NULL) 
			cout << i<<endl;
}
```
